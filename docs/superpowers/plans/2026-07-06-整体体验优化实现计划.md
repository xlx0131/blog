- [ ] **Step 1: 创建 Breadcrumb.vue 组件结构**

在 `src/components/Breadcrumb.vue` 中创建组件基础结构：

```vue
<template>
  <div class="breadcrumb" v-if="items.length > 0">
    <div class="breadcrumb-inner">
      <template v-for="(item, index) in items" :key="index">
        <span
          v-if="index < items.length - 1"
          class="breadcrumb-item clickable"
          @click="navigateTo(item.path)"
        >
          {{ item.name }}
        </span>
        <span v-else class="breadcrumb-item current">
          {{ item.name }}
        </span>
        <span v-if="index < items.length - 1" class="breadcrumb-sep">/</span>
      </template>
    </div>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue'
import { useRoute, useRouter } from 'vue-router'

const route = useRoute()
const router = useRouter()

const routeMap: Record<string, string> = {
  home: '首页',
  about: '关于',
  projects: '作品集',
  'project-detail': '项目详情',
  archive: '归档',
  article: '文章',
  bookmarks: '书签',
  'bookmark-detail': '书签详情',
  'network-game': '网络运维游戏',
  admin: '管理后台',
  writing: '创作',
}

const items = computed(() => {
  const result: { name: string; path: string }[] = []
  const name = route.name as string
  
  if (!name || name === 'home') return result

  result.push({ name: '首页', path: '/' })

  const parentMap: Record<string, { name: string; path: string }> = {
    'project-detail': { name: '作品集', path: '/projects' },
    article: { name: '归档', path: '/archive' },
    'bookmark-detail': { name: '书签', path: '/bookmarks' },
  }

  if (parentMap[name]) {
    result.push(parentMap[name])
  }

  const currentName = routeMap[name] || name
  result.push({ name: currentName, path: route.fullPath })

  return result
})

const navigateTo = (path: string) => {
  router.push(path)
}
</script>
```

### Task 2: 图片懒加载组件 LazyImage.vue

**Files:**
- Create: `src/components/LazyImage.vue`

- [ ] **Step 1: 创建 LazyImage.vue 组件**

```vue
<template>
  <div class="lazy-image" :style="{ aspectRatio: ratio }">
    <div v-if="!loaded && !error" class="placeholder">
      <div class="pixel-loader">
        <span v-for="i in 4" :key="i" class="pixel-dot" :style="{ animationDelay: (i - 1) * 0.15 + 's' }"></span>
      </div>
    </div>
    <div v-if="error" class="placeholder error">
      <span>加载失败</span>
    </div>
    <img
      ref="imgRef"
      :src="loaded ? src : ''"
      :alt="alt"
      :class="{ 'img-loaded': loaded }"
      @load="onLoad"
      @error="onError"
      loading="lazy"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted, onUnmounted } from 'vue'

const props = defineProps<{
  src: string
  alt?: string
  ratio?: number | string
}>()

const imgRef = ref<HTMLImageElement | null>(null)
const loaded = ref(false)
const error = ref(false)
let observer: IntersectionObserver | null = null

const onLoad = () => {
  loaded.value = true
}

const onError = () => {
  error.value = true
  loaded.value = false
}

onMounted(() => {
  if ('IntersectionObserver' in window) {
    observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            if (imgRef.value) {
              imgRef.value.src = props.src
            }
            observer?.disconnect()
          }
        })
      },
      { rootMargin: '100px' }
    )

    if (imgRef.value) {
      observer.observe(imgRef.value)
    }
  } else {
    if (imgRef.value) {
      imgRef.value.src = props.src
    }
  }
})

onUnmounted(() => {
  observer?.disconnect()
})
</script>
```

### Task 6: 文章目录组件 ArticleToc.vue

**Files:**

- [ ] **Step 1: 创建文章目录组件**

```vue
<template>
  <aside class="article-toc" v-if="headings.length > 0">
    <div class="toc-header">目录</div>
    <nav class="toc-list">
      <a
        v-for="(heading, index) in headings"
        :key="index"
        class="toc-item"
        :class="[`level-${heading.level}`, { active: activeIndex === index }]"
        @click.prevent="scrollToHeading(index)"
      >
        {{ heading.text }}
      </a>
    </nav>
  </aside>
</template>

<script setup lang="ts">
import { ref, onMounted, onUnmounted, nextTick } from 'vue'

const props = defineProps<{
  contentSelector?: string
}>()

interface Heading {
  level: number
  text: string
  element: HTMLElement
}

const headings = ref<Heading[]>([])
const activeIndex = ref(0)
let observer: IntersectionObserver | null = null

const extractHeadings = () => {
  const container = props.contentSelector
    ? document.querySelector(props.contentSelector)
    : document.querySelector('.article-content')

  if (!container) return

  const elements = container.querySelectorAll('h2, h3')
  headings.value = Array.from(elements).map((el) => ({
    level: parseInt(el.tagName.charAt(1)),
    text: el.textContent || '',
    element: el as HTMLElement,
  }))
}

const scrollToHeading = (index: number) => {
  const heading = headings.value[index]
  if (heading) {
    heading.element.scrollIntoView({ behavior: 'smooth', block: 'start' })
  }
}

const setupObserver = () => {
  if (!('IntersectionObserver' in window)) return
  if (headings.value.length === 0) return

  observer = new IntersectionObserver(
    (entries) => {
      const visibleEntries = entries
        .filter((entry) => entry.isIntersecting)
        .sort((a, b) => a.boundingClientRect.top - b.boundingClientRect.top)

      if (visibleEntries.length > 0) {
        const visibleEl = visibleEntries[0].target
        const index = headings.value.findIndex((h) => h.element === visibleEl)
        if (index !== -1) {
          activeIndex.value = index
        }
      }
    },
    {
      rootMargin: '-80px 0px -60% 0px',
      threshold: 1,
    }
  )

  headings.value.forEach((h) => {
    observer?.observe(h.element)
  })
}

onMounted(async () => {
  await nextTick()
  extractHeadings()
  setupObserver()
})

onUnmounted(() => {
  observer?.disconnect()
})
</script>
```
